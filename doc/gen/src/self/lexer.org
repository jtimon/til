struct Map__Str_TokenType {
    mut keys : Vec__Str
    mut values : Vec__TokenType
namespace:
    /// Create an empty Map of the types provided for the key and values respectively.
    /// The key type must implement `cmp`.
    func new() returns Map__Str_TokenType
    func len(self: Map__Str_TokenType) returns shallow U32
    /// Drop all live keys and values but keep the allocated storage so the Map can be reused.
    func clear(mut self: Map__Str_TokenType)
    /// Ref to the key slot at sorted index `i`. Unchecked: callers
    /// guarantee `i < self.keys.count` (Map's binary-search consumers
    /// always validate before calling). Bypasses Vec.get's throwing
    /// bounds check.
    func key_ptr(self: Map__Str_TokenType, i: U32) returns ref Str
    /// Ref to the value slot at sorted index `i`. Same caller-validated
    /// contract as key_ptr.
    func val_ptr(self: Map__Str_TokenType, i: U32) returns ref TokenType
    /// True if a value exists for the given key.
    func has(self: Map__Str_TokenType, key: Str) returns shallow Bool
    /// Return a ref to the value for `key`. Throws KeyNotFound if
    /// `key` is absent; call as `map.get(k)!` (or `map[k]`) to panic
    /// on a missing key.
    func get(self: Map__Str_TokenType, key: Str, mut _err_kind: I64, mut _err_KeyNotFound: KeyNotFound) returns ref TokenType
    /// Insert or overwrite the value for `key`. The value is owned.
    func set(mut self: Map__Str_TokenType, key: Str, own val: TokenType)
    func delete(own self: Map__Str_TokenType, shallow call_free: Bool)
    func clone(self: Map__Str_TokenType) returns Map__Str_TokenType
    func hash(self: Map__Str_TokenType, hasher: HashFn) returns shallow U64
    func size() returns shallow U32
}

#+begin_src til
struct Vec__Token {
    own mut data : U8
    mut count : U32
    mut cap : U32
namespace:
    /// Create a new empty Vec for elements of type T.
    func new() returns Vec__Token
    func len(self: Vec__Token) returns shallow U32
    /// Drop all live elements via elem_delete but keep the allocated
    /// storage and element-type metadata so the Vec can be reused.
    func clear(mut self: Vec__Token)
    /// Append an element to the end. `val` is owned: after this call the
    /// caller must not use the value passed -- clone it first if needed.
    func push(mut self: Vec__Token, own val: Token)
    /// Consume the contents of `other` by appending them to `self`.
    /// `other` is taken by ownership and will be empty after the call.
    func append(mut self: Vec__Token, own other: Vec__Token)
    /// Replace self's contents with other's via memcpy (no per-elem clone).
    /// Avoids the Vec_clone + extra Vec_delete that plain `self = other` emits.
    /// Takes `mut` (not `own`) so caller retains ownership of other; its
    /// natural scope-end auto-delete frees other.data. Setting other.count = 0
    /// prevents that auto-delete from re-running elem_delete on items whose
    /// bytes were moved into self.
    func move_from(mut self: Vec__Token, mut other: Vec__Token)
    /// Return a reference to the element at index `i`. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i]`
    /// to panic on a bad index. The reference is valid until the
    /// next mutation.
    func get(self: Vec__Token, i: U32, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds) returns ref Token
    /// Remove and return the last element. Panics if empty. The caller
    /// takes ownership of the returned value.
    func pop(mut self: Vec__Token) returns Token
    /// Split off the first `n` elements into a new Vec. Panics if `n`
    /// exceeds self.len(). The remaining elements shift to the front.
    func take_prefix(mut self: Vec__Token, shallow n: U32) returns Vec__Token
    /// Overwrite the element at index `i`. `val` is owned. Panics if `i`
    /// is out of bounds.
    func set(mut self: Vec__Token, shallow i: U32, own val: Token)
    /// Move bytes of src[i] into self (push). Source slot is zeroed so that
    /// src's eventual elem_delete on it is a no-op (relies on the same
    /// count==0/data==NULL convention used by take_children and Vec.delete).
    func push_take(mut self: Vec__Token, mut src: Vec__Token, shallow i: U32)
    /// Drop each element via elem_delete, free storage, then free self
    /// unless `call_free` is false.
    func delete(own self: Vec__Token, shallow call_free: Bool)
    /// Deep-copy the Vec and all its elements via elem_clone.
    func clone(self: Vec__Token) returns Vec__Token
    func size() returns shallow U32
}
