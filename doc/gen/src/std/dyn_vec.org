* DynVec
Dynamic array of typed elements.

#+begin_src til
struct DynVec {
    own mut data : U8
    mut count : U32
    mut cap : U32
    elem_size : U32
    elem_kind : U32
    elem_clone : CloneFn
    elem_delete : DeleteFn
namespace:
    /// Create a new empty DynVec for elements of the given type.
    func new_type_name(elem_type: Str) returns DynVec
    func new(T: Type) returns DynVec
    func len(self: DynVec) returns shallow U32
    /// Drop all live elements via elem_delete but keep the allocated
    /// storage and element-type metadata so the DynVec can be reused.
    func clear(mut self: DynVec)
    /// Append an element to the end. `val` is owned: after this call the
    /// caller must not use the value passed -- clone it first if needed.
    func push(mut self: DynVec, own val: Dynamic)
    /// Consume the contents of `other` by appending them to `self`.
    /// `other` is taken by ownership and will be empty after the call.
    func append(mut self: DynVec, own other: DynVec)
    /// Replace self's contents with other's via memcpy (no per-elem clone).
    /// Avoids the Vec_clone + extra Vec_delete that plain `self = other` emits.
    /// Takes `mut` (not `own`) so caller retains ownership of other; its
    /// natural scope-end auto-delete frees other.data. Setting other.count = 0
    /// prevents that auto-delete from re-running elem_delete on items whose
    /// bytes were moved into self.
    func move_from(mut self: DynVec, mut other: DynVec)
    /// Return a reference to the element at index `i`. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i]`
    /// to panic on a bad index. The reference is valid until the
    /// next mutation.
    func get(self: DynVec, i: U32, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds) returns ref Dynamic
    /// Remove and return the last element. Panics if empty. The caller
    /// takes ownership of the returned value.
    func pop(mut self: DynVec) returns Dynamic
    /// Split off the first `n` elements into a new DynVec. Panics if `n`
    /// exceeds self.len(). The remaining elements shift to the front.
    func take_prefix(mut self: DynVec, shallow n: U32) returns DynVec
    /// Overwrite the element at index `i`. `val` is owned. Panics if `i`
    /// is out of bounds.
    func set(mut self: DynVec, shallow i: U32, own val: Dynamic)
    /// Move bytes of src[i] into self (push). Source slot is zeroed so that
    /// src's eventual elem_delete on it is a no-op (relies on the same
    /// count==0/data==NULL convention used by take_children and DynVec.delete).
    func push_take(mut self: DynVec, mut src: DynVec, shallow i: U32)
    /// Drop each element via elem_delete, free storage, then free self
    /// unless `call_free` is false.
    func delete(own self: DynVec, shallow call_free: Bool)
    /// Deep-copy the DynVec and all its elements via elem_clone.
    func clone(self: DynVec) returns DynVec
    func size() returns shallow U32
}
#+end_src
