* Str
UTF-8 string with owned heap storage.

#+begin_src til
struct Str {
    own mut c_str : U8
    mut count : U32
    mut cap : U32
namespace:
    func len(self: Str) returns shallow U32
    /// Return a ref to byte `i`. Panics if `i` is out of bounds.
    func get(self: Str, i: U32) returns ref U8
    /// Unchecked byte access — caller must ensure i < self.len().
    func byte_at(self: Str, i: U32) returns ref U8
    /// Lexicographic comparison. Returns 0 if equal, negative if a < b,
    /// positive if a > b.
    func cmp(a: Str, b: Str) returns shallow I64
    /// Return a new Str with the contents of `a` followed by `b`.
    func concat(a: Str, b: Str) returns Str
    /// Create an empty Str pre-allocated for up to `n` bytes.
    func with_capacity(n: U32) returns Str
    /// Append `s` to `self`. Panics if `self` is a view/literal or if
    /// capacity is exceeded.
    func push_str(mut self: Str, s: Str)
    /// Deep-copy the string into a new owned Str.
    func clone(val: Str) returns Str
    /// Free the backing buffer (unless it's a view/literal) and optionally
    /// the struct allocation.
    func delete(own self: Str, call_free: Bool)
    /// Return a clone of the string.
    func to_str(val: Str) returns Str
    /// Return a view into `s` from `start` for `n` bytes (no allocation).
    /// Clamps to valid bounds.
    func substr(s: Str, start: U32, n: U32) returns Str
    /// True if `a` contains `b` as a substring.
    func contains(a: Str, b: Str) returns shallow Bool
    /// True if `a` starts with prefix `b`.
    func starts_with(a: Str, b: Str) returns shallow Bool
    /// True if `a` ends with suffix `b`.
    func ends_with(a: Str, b: Str) returns shallow Bool
    /// True if the string has zero length.
    func is_empty(self: Str) returns shallow Bool
    /// Return the byte index of the first occurrence of `needle`, or -1.
    func find(self: Str, needle: Str) returns shallow I64
    /// Return the byte index of the last occurrence of `needle`, or -1.
    func rfind(self: Str, needle: Str) returns shallow I64
    /// Return a new Str with all occurrences of `from` replaced by `to`.
    func replace(self: Str, from: Str, to: Str) returns Str
    /// View of the single byte at index `i` as a Str.
    func get_char(self: Str, i: U32) returns Str
    /// If self starts with `prefix`, return the rest; otherwise clone.
    func strip_prefix(self: Str, prefix: Str) returns Str
    /// If self ends with `suffix`, return the part before it; otherwise clone.
    func strip_suffix(self: Str, suffix: Str) returns Str
    /// Create a 1-byte Str from a U8 value.
    func from_byte(byte: U8) returns Str
    /// Parse the string as a base-10 signed integer. Panics on invalid input.
    func to_i64(self: Str) returns shallow I64
    /// Parse as base-10 unsigned 8-bit integer.
    func to_u8(self: Str) returns shallow U8
    /// Parse as base-10 signed 32-bit integer.
    func to_i32(self: Str) returns shallow I32
    /// Parse as a decimal floating-point number.
    func to_f32(self: Str) returns shallow F32
    /// Split the string on `delim`, returning a Vec of Str clones.
    func split(self: Str, delim: Str) returns Vec
    func size() returns shallow U32
    func eq(a: Str, b: Str) returns shallow Bool
    func lt(a: Str, b: Str) returns shallow Bool
    func gt(a: Str, b: Str) returns shallow Bool
    func neq(a: Str, b: Str) returns shallow Bool
    func lte(a: Str, b: Str) returns shallow Bool
    func gte(a: Str, b: Str) returns shallow Bool
}
#+end_src
