* Str
UTF-8 string with owned heap storage. The byte buffer is typed as
I8 (= C `char`) so direct FFI bindings against `const char *` APIs
can pass `s.c_str` without an intermediate pointer-cast helper.

#+begin_src til
struct Str {
    own mut c_str : I8
    mut count : U32
    mut cap : U32
namespace:
    func len(self: Str) returns shallow U32
    /// Return a ref to byte `i`. Panics if `i` is out of bounds.
    func get(self: Str, i: U32) returns ref I8
    /// Unchecked byte access -- caller must ensure i < self.len().
    func byte_at(self: Str, i: U32) returns ref I8
    /// Lexicographic comparison. Returns 0 if equal, negative if a < b,
    /// positive if a > b.
    func cmp(a: Str, b: Str) returns shallow I64
    /// Return a new Str with the contents of `a` followed by `b`.
    func concat(a: Str, b: Str) returns Str
    /// Create an empty Str pre-allocated for up to `n` bytes.
    func with_capacity(shallow n: U32) returns Str
    /// Wrap a NUL-terminated C string as a non-owning Str view (no copy).
    /// The view borrows `s`; the caller keeps ownership of the buffer.
    func from_c_str(s: Dynamic) returns Str
    /// Append `s` to `self`. Panics if `self` is a view/literal or if
    /// capacity is exceeded.
    func push_str(mut self: Str, s: Str)
    /// Deep-copy the string into a new owned Str.
    func clone(val: Str) returns Str
    /// Free the backing buffer (unless it's a view/literal) and optionally
    /// the struct allocation.
    func delete(own self: Str, shallow call_free: Bool)
    /// Return a view into `s` from `start` for `n` bytes (no allocation).
    /// Clamps to valid bounds.
    func substr(s: Str, shallow start: U32, shallow n: U32) returns Str
    /// Return a view of `s` with leading ASCII whitespace (space, tab,
    /// newline, carriage return) removed. View-only -- no allocation.
    func trim_start(s: Str) returns Str
    /// Return a view of `s` with leading and trailing ASCII whitespace
    /// removed. View-only -- no allocation.
    func trim(s: Str) returns Str
    /// True if `a` contains `b` as a substring.
    func contains(a: Str, b: Str) returns shallow Bool
    /// True if `a` starts with prefix `b`.
    func starts_with(a: Str, b: Str) returns shallow Bool
    /// True if `a` ends with suffix `b`.
    func ends_with(a: Str, b: Str) returns shallow Bool
    /// True if the string has zero length.
    func is_empty(self: Str) returns shallow Bool
    /// Return the byte index of the first occurrence of `needle`, or -1.
    func find(self: Str, needle: Str) returns shallow I64
    /// Return the byte index of the last occurrence of `needle`, or -1.
    func rfind(self: Str, needle: Str) returns shallow I64
    /// Return a new Str with all occurrences of `from` replaced by `to`.
    func replace(self: Str, from: Str, to: Str) returns Str
    /// View of the single byte at index `i` as a Str.
    func get_char(self: Str, shallow i: U32) returns Str
    /// Hash via the caller-supplied byte hasher, feeding it the
    /// string CONTENT bytes (c_str + count). The default bytewise
    /// fallback would hash the {c_str, count, cap} struct itself --
    /// i.e. the heap pointer's address -- which is identity, not
    /// equality. The container chooses the algorithm; this method
    /// chooses the right bytes for a Str.
    func hash(self: Str, hasher: HashFn) returns shallow U64
    /// If self starts with `prefix`, return the rest; otherwise clone.
    func strip_prefix(self: Str, prefix: Str) returns Str
    /// If self ends with `suffix`, return the part before it; otherwise clone.
    func strip_suffix(self: Str, suffix: Str) returns Str
    /// Create a 1-byte Str from a U8 value.
    func from_byte(shallow byte: U8) returns Str
    /// Parse the string as a signed integer. Accepts an optional leading
    /// '-', base prefixes 0x/0b/0o (issue #255), and '_' digit separators.
    /// Bare digit runs are base-10. Panics on invalid input.
    func to_i64(self: Str) returns shallow I64
    /// Parse as base-10 unsigned 8-bit integer.
    func to_u8(self: Str) returns shallow U8
    /// Parse as base-10 signed 8-bit integer.
    func to_i8(self: Str) returns shallow I8
    /// Parse as base-10 signed 16-bit integer.
    func to_i16(self: Str) returns shallow I16
    /// Parse as base-10 unsigned 16-bit integer.
    func to_u16(self: Str) returns shallow U16
    /// Parse as base-10 signed 32-bit integer.
    func to_i32(self: Str) returns shallow I32
    /// Parse as base-10 unsigned 32-bit integer.
    func to_u32(self: Str) returns shallow U32
    /// Parse as base-10 unsigned 64-bit integer.
    func to_u64(self: Str) returns shallow U64
    /// Parse as base-10 USize (collection sizes/indices).
    func to_usize(self: Str) returns shallow U32
    /// Parse as a floating-point number. Backed by C strtof: handles
    /// decimal and scientific notation at full precision, so a value
    /// printed by F32.to_repr round-trips exactly back to the same F32.
    /// '_' digit separators (issue #255) are stripped first since strtof
    /// does not accept them.
    func to_f32(self: Str) returns shallow F32
    /// Split the string on `delim`, returning a Vec of Str clones.
    func split(self: Str, delim: Str) returns Vec
    func size() returns shallow U32
    func eq(a: Str, b: Str) returns shallow Bool
    func lt(a: Str, b: Str) returns shallow Bool
    func gt(a: Str, b: Str) returns shallow Bool
    func neq(a: Str, b: Str) returns shallow Bool
    func lte(a: Str, b: Str) returns shallow Bool
    func gte(a: Str, b: Str) returns shallow Bool
}
#+end_src

* OutOfBounds
Error thrown by the indexed collection accessors (Array.get,
Vec.get, ...) when the index is past the end. The msg field
carries a "file:line:col:<method>: index N out of bounds (len M)"
description set at the throw site.

#+begin_src til
struct OutOfBounds {
    mut msg : Str
namespace:
    func clone(self: OutOfBounds) returns OutOfBounds
    func delete(own self: OutOfBounds, shallow call_free: Bool)
    func hash(self: OutOfBounds, hasher: HashFn) returns shallow U64
    func size() returns shallow U32
}
#+end_src

* KeyNotFound
Error type for keyed accessors (Map.get, HashMap.get, ...) when
the key is not present. The msg field carries a
"file:line:col:<method>: key not found" description set at the
throw site.

#+begin_src til
struct KeyNotFound {

    func delete(own self: KeyNotFound, shallow call_free: Bool)
    func hash(self: KeyNotFound, hasher: HashFn) returns shallow U64
    func size() returns shallow U32
}
