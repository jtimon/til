* Str
:PROPERTIES:
:CUSTOM_ID: til.src.core.str.Str
:group: Core
:END:

UTF-8 string with owned heap storage. The byte buffer is typed as
I8 (= C `char`) so direct FFI bindings against `const char *` APIs
can pass `s.c_str` without an intermediate pointer-cast helper.

#+begin_src til
struct Str {
    own mut c_str : I8
    mut count : U32
    mut cap : U32
namespace:
    func len(self: Str) returns shallow U32
    /// Return a ref to byte `i`. Panics if `i` is out of bounds.
    func get(self: Str, i: U32) returns ref I8
    /// Unchecked byte access -- caller must ensure i < self.len().
    func unsafe_get(self: Str, i: U32) returns ref I8
    /// Lexicographic comparison. Returns 0 if equal, negative if a < b,
    /// positive if a > b.
    func cmp(a: Str, b: Str) returns shallow I64
    /// Create an empty Str pre-allocated for up to `n` bytes.
    func with_capacity(shallow n: U32) returns Str
    /// Wrap a NUL-terminated C string as a non-owning Str view (no copy).
    /// The view borrows `s`; the caller keeps ownership of the buffer.
    func from_c_str(s: Dynamic) returns Str
    /// Append `s` to `self`. Panics if `self` is a view/literal or if
    /// capacity is exceeded.
    func push_str(mut self: Str, s: Str)
    /// Deep-copy the string into a new owned Str.
    func clone(val: Str) returns Str
    /// String representation of a Str: a fresh owned copy of itself. Lets
    /// Str satisfy the same `T.to_str(x)` contract as the numeric types, so
    /// generic code (e.g. assert_eq(Str, a, b)) works uniformly.
    func to_str(self: Str) returns Str
    /// Free the backing buffer unless it is borrowed/static. CAP_LIT names a
    /// static literal wrapper, so call_free does not free that wrapper either.
    func delete(own self: Str, shallow call_free: Bool)
    /// Return a view into `s` from `start` for `n` bytes (no allocation).
    /// Clamps to valid bounds.
    func substr(s: Str, shallow start: U32, shallow n: U32) returns Str
    /// Return a view of `s` with leading ASCII whitespace (space, tab,
    /// newline, carriage return) removed. View-only -- no allocation.
    func trim_start(s: Str) returns Str
    /// Return a view of `s` with leading and trailing ASCII whitespace
    /// removed. View-only -- no allocation.
    func trim(s: Str) returns Str
    /// True if `a` contains `b` as a substring.
    func contains(a: Str, b: Str) returns shallow Bool
    /// True if `a` starts with prefix `b`.
    func starts_with(a: Str, b: Str) returns shallow Bool
    /// True if `a` ends with suffix `b`.
    func ends_with(a: Str, b: Str) returns shallow Bool
    /// True if the string has zero length.
    func is_empty(self: Str) returns shallow Bool
    /// Return the byte index of the first occurrence of `needle`, or -1.
    func find(self: Str, needle: Str) returns shallow I64
    /// Return the byte index of the last occurrence of `needle`, or -1.
    func rfind(self: Str, needle: Str) returns shallow I64
    /// Return a new Str with all occurrences of `from` replaced by `to`.
    func replace(self: Str, from: Str, to: Str) returns Str
    /// View of the single byte at index `i` as a Str.
    func get_char(self: Str, shallow i: U32) returns Str
    /// Hash via the caller-supplied byte hasher, feeding it the
    /// string CONTENT bytes (c_str + count). The default bytewise
    /// fallback would hash the {c_str, count, cap} struct itself --
    /// i.e. the heap pointer's address -- which is identity, not
    /// equality. The container chooses the algorithm; this method
    /// chooses the right bytes for a Str.
    func hash(self: Str, hasher: HashFn) returns shallow U64
    /// If self starts with `prefix`, return the rest; otherwise clone.
    func strip_prefix(self: Str, prefix: Str) returns Str
    /// If self ends with `suffix`, return the part before it; otherwise clone.
    func strip_suffix(self: Str, suffix: Str) returns Str
    /// Create a 1-byte Str from a U8 value.
    func from_byte(shallow byte: U8) returns Str
    /// Parse the string as a signed integer. Accepts an optional leading
    /// '-', base prefixes 0x/0b/0o (issue #255), and '_' digit separators.
    /// Bare digit runs are base-10. Panics on invalid input.
    func to_i64(self: Str) returns shallow I64
    /// Parse as base-10 unsigned 8-bit integer.
    func to_u8(self: Str) returns shallow U8
    /// Parse as base-10 signed 8-bit integer.
    func to_i8(self: Str) returns shallow I8
    /// Parse as base-10 signed 16-bit integer.
    func to_i16(self: Str) returns shallow I16
    /// Parse as base-10 unsigned 16-bit integer.
    func to_u16(self: Str) returns shallow U16
    /// Parse as base-10 signed 32-bit integer.
    func to_i32(self: Str) returns shallow I32
    /// Parse as base-10 unsigned 32-bit integer.
    func to_u32(self: Str) returns shallow U32
    /// Parse as base-10 unsigned 64-bit integer.
    func to_u64(self: Str) returns shallow U64
    /// Parse as base-10 USize (collection sizes/indices).
    func to_usize(self: Str) returns shallow U32
    /// Parse as a floating-point number. Backed by C strtof: handles
    /// decimal and scientific notation at full precision, so a value
    /// printed by F32.to_repr round-trips exactly back to the same F32.
    /// '_' digit separators (issue #255) are stripped first since strtof
    /// does not accept them.
    func to_f32(self: Str) returns shallow F32
    /// Split the string on `delim`, returning a Vec of Str clones.
    func split(self: Str, delim: Str) returns Vec__Str
    func size() returns shallow U32
    func eq(a: Str, b: Str) returns shallow Bool
    func lt(a: Str, b: Str) returns shallow Bool
    func gt(a: Str, b: Str) returns shallow Bool
    func neq(a: Str, b: Str) returns shallow Bool
    func lte(a: Str, b: Str) returns shallow Bool
    func gte(a: Str, b: Str) returns shallow Bool
}

struct Array__Str {
    own mut data : U8
    cap : U32
namespace:
    /// Create a zero-initialized Array with the given capacity.
    func new(shallow cap: U32) returns Array__Str
    func len(self: Array__Str) returns shallow U32
    /// Unchecked element access: ref to the element at index `i` with
    /// NO bounds check. The caller must prove `i < len()`. This is the
    /// unchecked core that `get` wraps; for-in uses it directly because
    /// its `_fi < len()` termination test already proves the index in
    /// bounds (issue #284).
    func unsafe_get(self: Array__Str, i: U32) returns ref Str
    /// Ref to the element at index `i`. Throws OutOfBounds if `i` is
    /// past the end; call as `arr.get(i)!` to panic on a bad index.
    func get(self: Array__Str, i: U32, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str) returns ref Str
    /// Overwrite the element at index `i` without a bounds check, dropping
    /// the old value. Caller guarantees `i < self.cap`; the write-side
    /// mirror of unsafe_get (issue #284).
    func unsafe_set(self: Array__Str, shallow i: U32, own val: Str)
    /// Overwrite the element at index `i`. Drops the old value. `val` is
    /// owned. Throws OutOfBounds if `i` is past the end; call as
    /// `arr[i] = val` (or `arr.set(i, own val)!`) to panic on a bad index.
    func set(self: Array__Str, shallow i: U32, own val: Str, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str)
    func delete(own self: Array__Str, shallow call_free: Bool)
    func clone(self: Array__Str) returns Array__Str
    func size() returns shallow U32
}

struct Vec__Str {
    own mut data : U8
    mut count : U32
    mut cap : U32
namespace:
    /// Create a new empty Vec for elements of type T.
    func new() returns Vec__Str
    func len(self: Vec__Str) returns shallow U32
    /// Drop all live elements via elem_delete but keep the allocated
    /// storage and element-type metadata so the Vec can be reused.
    func clear(mut self: Vec__Str)
    /// Append an element to the end. `val` is owned: after this call the
    /// caller must not use the value passed -- clone it first if needed.
    func push(mut self: Vec__Str, own val: Str)
    /// Consume the contents of `other` by appending them to `self`.
    /// `other` is taken by ownership and will be empty after the call.
    func append(mut self: Vec__Str, own other: Vec__Str)
    /// Replace self's contents with other's via memcpy (no per-elem clone).
    /// Avoids the Vec_clone + extra Vec_delete that plain `self = other` emits.
    /// Takes `mut` (not `own`) so caller retains ownership of other; its
    /// natural scope-end auto-delete frees other.data. Setting other.count = 0
    /// prevents that auto-delete from re-running elem_delete on items whose
    /// bytes were moved into self.
    func move_from(mut self: Vec__Str, mut other: Vec__Str)
    /// Unchecked element access: return a reference to the element at
    /// index `i` with NO bounds check. The caller must prove `i < len()`.
    /// This is the unchecked core that `get` wraps; for-in uses it
    /// directly because its `_fi < len()` termination test already
    /// proves the index in bounds (issue #284).
    func unsafe_get(self: Vec__Str, i: U32) returns ref Str
    /// Return a reference to the element at index `i`. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i]`
    /// to panic on a bad index. The reference is valid until the
    /// next mutation.
    func get(self: Vec__Str, i: U32, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str) returns ref Str
    /// Remove and return the last element. Panics if empty. The caller
    /// takes ownership of the returned value.
    func pop(mut self: Vec__Str) returns Str
    /// Split off the first `n` elements into a new Vec. Panics if `n`
    /// exceeds self.len(). The remaining elements shift to the front.
    func take_prefix(mut self: Vec__Str, shallow n: U32) returns Vec__Str
    /// Overwrite the element at index `i` without a bounds check, dropping
    /// the old value. Caller guarantees `i < self.count`; the write-side
    /// mirror of unsafe_get (issue #284).
    func unsafe_set(mut self: Vec__Str, shallow i: U32, own val: Str)
    /// Overwrite the element at index `i`. `val` is owned. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i] = val`
    /// (or `vec.set(i, own val)!`) to panic on a bad index.
    func set(mut self: Vec__Str, shallow i: U32, own val: Str, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str)
    /// Move bytes of src[i] into self (push). Source slot is zeroed so that
    /// src's eventual elem_delete on it is a no-op (relies on the same
    /// count==0/data==NULL convention used by take_children and Vec.delete).
    func push_take(mut self: Vec__Str, mut src: Vec__Str, shallow i: U32)
    /// Drop each element via elem_delete, free storage, then free self
    /// unless `call_free` is false.
    func delete(own self: Vec__Str, shallow call_free: Bool)
    /// Deep-copy the Vec and all its elements via elem_clone.
    func clone(self: Vec__Str) returns Vec__Str
    func size() returns shallow U32
}
#+end_src
