struct Vec__Tuple {
    own mut data : U8
    mut count : U32
    mut cap : U32
namespace:
    /// Create a new empty Vec for elements of type T.
    func new() returns Vec__Tuple
    func len(self: Vec__Tuple) returns shallow U32
    /// Drop all live elements via elem_delete but keep the allocated
    /// storage and element-type metadata so the Vec can be reused.
    func clear(mut self: Vec__Tuple)
    /// Append an element to the end. `val` is owned: after this call the
    /// caller must not use the value passed -- clone it first if needed.
    func push(mut self: Vec__Tuple, own val: Tuple)
    /// Consume the contents of `other` by appending them to `self`.
    /// `other` is taken by ownership and will be empty after the call.
    func append(mut self: Vec__Tuple, own other: Vec__Tuple)
    /// Replace self's contents with other's via memcpy (no per-elem clone).
    /// Avoids the Vec_clone + extra Vec_delete that plain `self = other` emits.
    /// Takes `mut` (not `own`) so caller retains ownership of other; its
    /// natural scope-end auto-delete frees other.data. Setting other.count = 0
    /// prevents that auto-delete from re-running elem_delete on items whose
    /// bytes were moved into self.
    func move_from(mut self: Vec__Tuple, mut other: Vec__Tuple)
    /// Unchecked element access: return a reference to the element at
    /// index `i` with NO bounds check. The caller must prove `i < len()`.
    /// This is the unchecked core that `get` wraps; for-in uses it
    /// directly because its `_fi < len()` termination test already
    /// proves the index in bounds (issue #284).
    func unsafe_get(self: Vec__Tuple, i: U32) returns ref Tuple
    /// Return a reference to the element at index `i`. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i]`
    /// to panic on a bad index. The reference is valid until the
    /// next mutation.
    func get(self: Vec__Tuple, i: U32, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str) returns ref Tuple
    /// Remove and return the last element. Panics if empty. The caller
    /// takes ownership of the returned value.
    func pop(mut self: Vec__Tuple) returns Tuple
    /// Split off the first `n` elements into a new Vec. Panics if `n`
    /// exceeds self.len(). The remaining elements shift to the front.
    func take_prefix(mut self: Vec__Tuple, shallow n: U32) returns Vec__Tuple
    /// Overwrite the element at index `i`. `val` is owned. Throws
    /// OutOfBounds if `i` is past the end; call as `vec[i] = val`
    /// (or `vec.set(i, own val)!`) to panic on a bad index.
    func set(mut self: Vec__Tuple, shallow i: U32, own val: Tuple, mut _err_kind: I64, mut _err_OutOfBounds: OutOfBounds, loc: Str)
    /// Move bytes of src[i] into self (push). Source slot is zeroed so that
    /// src's eventual elem_delete on it is a no-op (relies on the same
    /// count==0/data==NULL convention used by take_children and Vec.delete).
    func push_take(mut self: Vec__Tuple, mut src: Vec__Tuple, shallow i: U32)
    /// Drop each element via elem_delete, free storage, then free self

    func delete(own self: Vec__Tuple, shallow call_free: Bool)
    /// Deep-copy the Vec and all its elements via elem_clone.
    func clone(self: Vec__Tuple) returns Vec__Tuple
