* DivideByZero
Error thrown by I64.safe_div / F32.safe_div when dividing by zero.
Carries a "file:line:col:<method>: divide by zero" msg pointing at
the caller (loc defaults to the call site, like IntegerOverflow).

#+begin_src til
struct DivideByZero {
    mut msg : Str
namespace:
    func clone(self: DivideByZero) returns DivideByZero
    func delete(own self: DivideByZero, shallow call_free: Bool)
    func hash(self: DivideByZero, hasher: HashFn) returns shallow U64
    func eq(self: DivideByZero, other: DivideByZero) returns shallow Bool
    func to_str(self: DivideByZero) returns Str
    func size() returns shallow U32
}
#+end_src

* IntegerOverflow
Error thrown by safe_add / safe_sub / safe_mul on the integer
types when the operation would wrap. Carries a
"file:line:col:<method>: a OP b overflows/underflows" msg. One
shared type for all integer widths.

#+begin_src til
struct IntegerOverflow {
    mut msg : Str
namespace:
    func clone(self: IntegerOverflow) returns IntegerOverflow
    func delete(own self: IntegerOverflow, shallow call_free: Bool)
    func hash(self: IntegerOverflow, hasher: HashFn) returns shallow U64
    func eq(self: IntegerOverflow, other: IntegerOverflow) returns shallow Bool
    func to_str(self: IntegerOverflow) returns Str
    func size() returns shallow U32
}
#+end_src
