* FCallData
Per-callsite data for function calls: parse-time flags (is_splat,
does_throw, is_bang), own_args bitmap, result type. The variadic
and kwargs block positions and counts are not stored -- they all
derive from the resolved callee's param layout plus the structural
markers in fcall.children:
  - kwargs_count : trailing-NamedArg count (fcall_kwargs_count).
  - kwargs_index : derived position via fcall_kwargs_index_via_scope.
  - variadic_count : children layout + nparam + has-kwargs
    (fcall_variadic_count).
  - variadic_index : implicit at `1 + fdef_data.variadic_index`
    post-desugar. Step 7 of #147 closes here: the trailing typer
    resolution artifact is gone.

#+begin_src til
struct FCallData {
    mut is_splat : Bool
    mut does_throw : Bool
    mut is_bang : Bool
    mut own_args : U64
    mut til_type : Type
namespace:

    func to_str(_self: FCallData) returns Str
    func clone(self: FCallData) returns FCallData
    func delete(own self: FCallData, shallow call_free: Bool)
    func hash(self: FCallData, hasher: HashFn) returns shallow U64

* Expr
AST expression node with line/col, type, children, and node-specific payload.

#+begin_src til
struct Expr {
    mut node_type : NodeType
    mut children : Vec
    mut line : U32
    mut col : U32
namespace:
    /// Print a diagnostic error with file path and source location.
    proc error(self: Expr, path: Str, msg: Str)
    /// Like error() but adds "not implemented yet" note.
    proc todo_error(self: Expr, path: Str, msg: Str)
    /// Like error() but adds "bug in the language" note.
    proc lang_error(self: Expr, path: Str, msg: Str)
    /// Take ownership of `child` and append it to the children list.
    func add_child(mut self: Expr, own child: Expr)
    /// Clone `child` and append the clone to the children list.
    func push_child_clone(mut self: Expr, child: Expr)
    /// Append other's children into self via raw memcpy (no cloning).
    /// Zeroes other's count so it no longer owns the moved items.
    func take_children(mut self: Expr, mut other: Expr)
    /// Shortcut to get a ref to the i-th child of `parent`.
    func child(parent: Expr, i: U32) returns ref Expr
    /// Create a new Expr with the given node_type and source location.
    func new(data: NodeType, shallow line: U32, shallow col: U32) returns Expr
    func eq(_self: Expr, _other: Expr) returns shallow Bool
    func to_str(self: Expr) returns Str
    func clone(self: Expr) returns Expr
    func delete(own self: Expr, shallow call_free: Bool)
    func hash(self: Expr, hasher: HashFn) returns shallow U64
    func size() returns shallow U32
}
