* TilType
Type representation for the til compiler. Carries payload for named
types (Struct/Enum/Custom) and is tag-only for builtins.

#+begin_src til
enum TilType {
    Unknown
    None
    I64
    U8
    I16
    I32
    U32
    U64
    F32
    Bool
    Struct: Str
    StructDef
    Enum: Str
    EnumDef
    FuncDef
    FuncPtr
    Dynamic
    Custom: Str
namespace:
    /// Compare two TilTypes. Named-payload variants (Struct/Enum/Custom)
    /// compare equal regardless of variant if their name payloads match.
    func eq(a: TilType, b: TilType) returns shallow Bool
    ext_func Unknown() returns TilType
    ext_func None() returns TilType
    ext_func I64() returns TilType
    ext_func U8() returns TilType
    ext_func I16() returns TilType
    ext_func I32() returns TilType
    ext_func U32() returns TilType
    ext_func U64() returns TilType
    ext_func F32() returns TilType
    ext_func Bool() returns TilType
    ext_func Struct(val: Str) returns TilType
    ext_func StructDef() returns TilType
    ext_func Enum(val: Str) returns TilType
    ext_func EnumDef() returns TilType
    ext_func FuncDef() returns TilType
    ext_func FuncPtr() returns TilType
    ext_func Dynamic() returns TilType
    ext_func Custom(val: Str) returns TilType
    func is(self: TilType, other: TilType) returns shallow Bool
    func delete(own self: TilType, call_free: Bool)
    func to_str(self: TilType) returns Str
    func clone(self: TilType) returns TilType
    func size() returns shallow U32
}

* Expr
AST expression node with line/col, type, children, and node-specific payload.

#+begin_src til
struct Expr {
    mut node_type : NodeType
    mut til_type : TilType
    mut children : Vec
    mut line : U32
    mut col : U32
namespace:
    /// Print a diagnostic error with file path and source location.
    proc error(self: Expr, path: Str, msg: Str)
    /// Like error() but adds "not implemented yet" note.
    proc todo_error(self: Expr, path: Str, msg: Str)
    /// Like error() but adds "bug in the language" note.
    proc lang_error(self: Expr, path: Str, msg: Str)
    /// Take ownership of `child` and append it to the children list.
    func add_child(mut self: Expr, own child: Expr)
    /// Clone `child` and append the clone to the children list.
    func push_child_clone(mut self: Expr, child: Expr)
    /// Append other's children into self via raw memcpy (no cloning).
    /// Zeroes other's count so it no longer owns the moved items.
    func take_children(mut self: Expr, mut other: Expr)
    /// Shortcut to get a ref to the i-th child of `parent`.
    func child(parent: Expr, i: U32) returns ref Expr
    /// Create a new Expr with the given node_type and source location.
    func new(data: NodeType, shallow line: U32, shallow col: U32) returns Expr
    func eq(_self: Expr, _other: Expr) returns shallow Bool
    func to_str(self: Expr) returns Str
    func clone(self: Expr) returns Expr
    func delete(own self: Expr, call_free: Bool)
    func size() returns shallow U32
}
