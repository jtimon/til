* DynMap
String-keyed map of dynamically-typed values.

#+begin_src til
struct DynMap {
    mut items : Map
namespace:
    /// Create an empty DynMap.
    func new() returns DynMap
    func len(self: DynMap) returns shallow U32

    func type_at(self: DynMap, key: Str) returns ref Str
    /// Byte size of the value stored under `key`.
    func size_at(self: DynMap, key: Str) returns ref U32
    func clone(self: DynMap) returns DynMap
    func delete(own self: DynMap, shallow call_free: Bool)
    func hash(self: DynMap, hasher: HashFn) returns shallow U64
    func size() returns shallow U32
}
