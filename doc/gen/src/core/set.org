* Set
Unordered collection of unique typed elements.

#+begin_src til
struct Set {
    own mut data : U8
    mut count : U32
    mut cap : U32
    elem_size : U32
    elem_type : Str
    elem_clone : CloneFn
    elem_delete : DeleteFn
    elem_cmp : CmpFn
namespace:
    /// Create an empty Set. elem_type must implement `cmp`.
    func new(T: Type) returns Set
    func len(self: Set) returns U32
    /// True if the set contains an element equal to `val`.
    func has(self: Set, val: Dynamic) returns shallow Bool
    /// Insert `val` into the set (sorted order). If an equal element
    /// already exists, `val` is dropped. `val` is owned.
    func add(mut self: Set, own val: Dynamic)
    func delete(own self: Set, call_free: Bool)
    func clone(self: Set) returns Set
    func size() returns shallow U32
}
#+end_src
