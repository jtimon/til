** OPEN #36 Nested pattern matching in switch

Depends on [[./12.org][#12]] (self-hosting) — not worth implementing before self-hosting.

  Full pattern matching as in rstil — nested enum patterns, struct field
  matching, wildcard ~_~. Phases 1-4 of [[./35.org][#35]] cover most practical use cases;
  this adds the remaining rstil features.

  Examples from ~../rstil/src/test/enums.til~:

  Nested payload extraction (currently requires nested switch):
  #+begin_src til
  // rstil supported this in a single switch:
  switch outer {
  case OuterEnum.WrapperA(inner):
      switch inner {
      case InnerEnum.ValueA(val):
          // use val
      case InnerEnum.ValueB(s):
          // use s
      }
  case OuterEnum.WrapperB(i):
      // use i
  }
  #+end_src

  Multi-variant function with payload (rstil test_color_match):
  #+begin_src til
  test_color_match := func(c: Color) returns Str {
      switch c {
      case Color.Unknown: return "unknown"
      case Color.Green(is_olive):
          if is_olive { return "olive" }
          return "dark green"
      case Color.Number(code):
          if code.gt(0) { return "positive" }
          return "negative"
      case Color.Name(name): return format("named: ", name)
      case Color.Rgb(rgb):
          return format("rgb(", rgb.r.to_str(), ",", rgb.g.to_str(), ",", rgb.b.to_str(), ")")
      }
  }
  #+end_src
